        // Carousel Logic
        const carouselState = {};

        function moveCarousel(trackId, direction) {
            const track = document.getElementById(trackId);
            if (!carouselState[trackId]) carouselState[trackId] = 0;
            
            const items = track.children.length;
            carouselState[trackId] += direction;
            
            if (carouselState[trackId] >= items) carouselState[trackId] = 0;
            if (carouselState[trackId] < 0) carouselState[trackId] = items - 1;
            
            track.style.transform = `translateX(-${carouselState[trackId] * 100}%)`;
        }
